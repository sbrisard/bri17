148  void modal_strain_displacement(
149  size_t const *k, Eigen::Matrix<std::complex<double>, DIM, 1> &B) const {
150  double c[DIM];
151  double s[DIM];
152  double sum_alpha = 0.;
153 
154  for (size_t i = 0; i < DIM; i++) {
155  const double alpha = M_PI * k[i] / grid.N[i];
156  sum_alpha+= alpha;
157  c[i] = cos(alpha) * grid.L[i] / grid.N[i];
158  s[i] = sin(alpha);
159  }
160 
161  std::complex prefactor{-2 * sin(sum_alpha), 2 * cos(sum_alpha)};
162 
163  if constexpr (DIM == 2) {
164  B(0) = prefactor * s[0] * c[1];
165  B(1) = prefactor * c[0] * s[1];
166  } else if constexpr (DIM == 3) {
167  B(0) = prefactor * s[0] * c[1] * c[2];
168  B(1) = prefactor * c[0] * s[1] * c[2];
169  B(2) = prefactor * c[0] * c[1] * s[2];

175  void modal_stiffness(size_t const *k,
176  Eigen::Matrix<std::complex<double>, DIM, DIM> &K) const {
177  // In the notation of [Bri17, see Eq. (B.17)]
178  //
179  // phi[i] = phi(z_i) / h_i
180  // chi[i] = chi(z_i) * h_i
181  // psi[i] = psi(z_i)
182  //
183  // Which simplifies the expression of H_k (there are no h_i's).
184  // Note that H_k is multiplied by the cell volume, so that the
185  // modal_stiffness is the true modal stiffness
186  double phi[DIM];
187  double psi[DIM];
188  double chi[DIM];
189  for (size_t i = 0; i < DIM; i++) {
190  double h = grid.L[i] / grid.N[i];
191  double beta = 2 * M_PI * k[i] / grid.N[i];
192  phi[i] = 2 * (1 - cos(beta)) / h;
193  chi[i] = (2 + cos(beta)) * h / 3;
194  psi[i] = sin(beta);
195  }
196 
197  const double scaling = mu / (1. - 2. * nu);
198  if constexpr (DIM == 2) {
199  const double H_00 = phi[0] * chi[1];
200  K(0, 0) = scaling * H_00;
201  K(0, 1) = scaling * psi[0] * psi[1];
202  const double H_11 = chi[0] * phi[1];
203  K(1, 1) = scaling * H_11;
204 
205  // Symmetrization
206  K(1, 0) = K(0, 1);
207 
208  const double K_diag = mu * (H_00 + H_11);
209  K(0, 0) += K_diag;
210  K(1, 1) += K_diag;
211  } else if constexpr (DIM == 3) {
212  const double H_00 = phi[0] * chi[1] * chi[2];
213  K(0, 0) = scaling * H_00;
214  K(0, 1) = scaling * psi[0] * psi[1] * chi[2];
215  K(0, 2) = scaling * psi[0] * chi[1] * psi[2];
216  const double H_11 = chi[0] * phi[1] * chi[2];
217  K(1, 1) = scaling * H_11;
218  K(1, 2) = scaling * chi[0] * psi[1] * psi[2];
219  const double H_22 = chi[0] * chi[1] * phi[2];
220  K(2, 2) = scaling * H_22;
221 
222  K(1, 0) = K(0, 1);
223  K(2, 0) = K(0, 2);
224  K(2, 1) = K(1, 2);
225 
226  const double K_diag = mu * (H_00 + H_11 + H_22);
227  K(0, 0) += K_diag;
228  K(1, 1) += K_diag;
229  K(2, 2) += K_diag;
230  } else {
231  throw std::logic_error("this should never occur");
232  }
233  }

236  size_t const *k, Eigen::Matrix<std::complex<double>, DIM, DIM> &tau,
